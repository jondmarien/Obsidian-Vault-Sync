15/// Looks up a user by Clerk user ID or email, creates the user if not found, and checks if their Devpost data has already been scraped.
16///
17/// # Arguments
18/// * `http_client` - The HTTP client for making requests to Supabase.
19/// * `supabase_url` - The Supabase instance URL.
20/// * `supabase_key` - The Supabase service role key.
21/// * `clerk_uid` - The Clerk user ID to look up.
22/// * `email` - The user's email address to look up.
23///
24/// # Returns
25/// * `Result<DevpostUserLookupResult>` - The result containing user lookup information or an error.
26pub async fn lookup_or_create_user_and_check_scraped(
27    http_client: &Client,
28    supabase_url: &str,
29    supabase_key: &str,
30    clerk_uid: &str,
31    email: &str,
32) -> anyhow::Result<DevpostUserLookupResult> {
33    // 1. Try to find by clerk_user_id
34    let user_query_url = format!("{}/rest/v1/user", supabase_url.trim_end_matches('/'));
35    let user_query_resp = http_client
36        .get(&user_query_url)
37        .header("apikey", supabase_key)
38        .header("Authorization", format!("Bearer {}", supabase_key))
39        .header("Content-Type", "application/json")
40        .query(&[
41            ("clerk_uid".to_string(), clerk_uid.to_string()),
42            ("select".to_string(), "id".to_string()),
43        ])
44        .send()
45        .await?;
46    let users_json: serde_json::Value = user_query_resp.json().await?;
47    let user_uuid = if let Some(uuid_val) = users_json
48        .get(0)
49        .and_then(|u| u.get("id"))
50        .and_then(|id| id.as_str()) {
51        uuid_val.to_string()
52    } else {
53
54        // 2. If not found, try to find by email
55        let email_query_resp = http_client
56            .get(&user_query_url)
57            .header("apikey", supabase_key)
58            .header("Authorization", format!("Bearer {}", supabase_key))
59            .header("Content-Type", "application/json")
60            .query(&[
61                ("email".to_string(), format!("eq.{}", email)),
62                ("select".to_string(), "id".to_string()),
63            ])
64            .send()
65            .await?;
66        let email_users_json: serde_json::Value = email_query_resp.json().await?;
67        if let Some(uuid_val) = email_users_json
68            .get(0)
69            .and_then(|u| u.get("id"))
70            .and_then(|id| id.as_str()) {
71            uuid_val.to_string()
72        } else {
73
74            // 3. If not found by either, insert a new user row
75            let insert_user_resp = http_client
76                .post(&user_query_url)
77                .header("apikey", supabase_key)
78                .header("Authorization", format!("Bearer {}", supabase_key))
79                .header("Content-Type", "application/json")
80                .header("Prefer", "return=representation")
81                .json(&serde_json::json!({
82                    "id": Uuid::new_v4().to_string(),
83                    "clerk_uid": clerk_uid,
84                    "email": email,
85                    "first_name": "Unknown" // Default value for required field
86                }))
87                .send()
88                .await?;
89            
90            // Check status code first
91            let status = insert_user_resp.status();
92            if !status.is_success() {
93                let error_text = insert_user_resp.text().await.unwrap_or_else(|_| "<failed to read response>".to_string());
94                return Err(anyhow::anyhow!("Failed to insert user: HTTP {}: {}", status, error_text));
95            }
96
97            // Parse response
98            let text = insert_user_resp.text().await.unwrap_or_else(|_| "<no body>".to_string());
99            let parse_result = serde_json::from_str::<serde_json::Value>(&text);
100            if let Err(e) = &parse_result {
101                return Err(anyhow::anyhow!("Invalid response from Supabase: {}, Response body: {}", e, text));
102            }
103            
104            let json_value = parse_result.unwrap();
105            
106            // Try to get ID - handle both array and single object formats
107            let user_id = if json_value.is_array() {
108                // Array format: [{"id": "..."}]
109                json_value.get(0)
110                    .and_then(|u| u.get("id"))
111                    .and_then(|id| id.as_str())
112            } else {
113                // Single object format: {"id": "..."}
114                json_value.get("id").and_then(|id| id.as_str())
115            };
116
117            // Return the ID or an error with context
118            user_id.map(|s| s.to_string()).ok_or_else(|| {
119                anyhow::anyhow!("Failed to get id from inserted user. Response: {}", text)
120            })?
121        }
122    };
123
124    // Check devpost_skills
125    let skills_exists_resp = http_client
126        .get(format!("{}/rest/v1/devpost_skills", supabase_url.trim_end_matches('/')))
127        .header("apikey", supabase_key)
128        .header("Authorization", format!("Bearer {}", supabase_key))
129        .header("Content-Type", "application/json")
130        .query(&[("user_uuid", format!("eq.{}", user_uuid)), ("select", "user_uuid".to_string())])
131        .send()
132        .await;
133    let skills_exists = if let Ok(resp) = skills_exists_resp {
134        if let Ok(json) = resp.json::<serde_json::Value>().await {
135            json.get(0).is_some()
136        } else { false }
137    } else { false };
138    
139    // Check devpost_interests
140    let interests_exists_resp = http_client
141        .get(format!("{}/rest/v1/devpost_interests", supabase_url.trim_end_matches('/')))
142        .header("apikey", supabase_key)
143        .header("Authorization", format!("Bearer {}", supabase_key))
144        .header("Content-Type", "application/json")
145        .query(&[("user_uuid", format!("eq.{}", user_uuid)), ("select", "user_uuid".to_string())])
146        .send()
147        .await;
148    let interests_exists = if let Ok(resp) = interests_exists_resp {
149        if let Ok(json) = resp.json::<serde_json::Value>().await {
150            json.get(0).is_some()
151        } else { false }
152    } else { false };
153    // Create the lookup result with all fields
154    let mut result = DevpostUserLookupResult {
155        user_uuid,
156        already_scraped: skills_exists && interests_exists,
157        ..Default::default()
158    };
159
160    // If data exists, fetch the additional details
161    if result.already_scraped {
162        // Fetch user details from the database
163        let user_details_url = format!(
164            "{}/rest/v1/user?id=eq.{}",
165            supabase_url.trim_end_matches('/'),
166            result.user_uuid
167        );
168        
169        if let Ok(resp) = http_client
170            .get(&user_details_url)
171            .header("apikey", supabase_key)
172            .header("Authorization", format!("Bearer {}", supabase_key))
173            .header("Accept", "application/vnd.pgrst.object+json")
174            .send()
175            .await
176        {
177            if let Ok(user_data) = resp.json::<serde_json::Value>().await {
178                result.name = user_data.get("first_name")
179                    .and_then(|n| n.as_str())
180                    .map(|s| s.to_string());
181                
182                // If we have a last name, append it
183                if let Some(last_name) = user_data.get("last_name")
184                    .and_then(|n| n.as_str())
185                    .filter(|s| !s.is_empty())
186                {
187                    result.name = result.name
188                        .map(|n| format!("{} {}", n, last_name))
189                        .or_else(|| Some(last_name.to_string()));
190                }
191            }
192        }
193        
194        // Fetch skills
195        let skills_url = format!(
196            "{}/rest/v1/devpost_skills?user_uuid=eq.{}",
197            supabase_url.trim_end_matches('/'),
198            result.user_uuid
199        );
200        
201        if let Ok(skills_resp) = http_client
202            .get(&skills_url)
203            .header("apikey", supabase_key)
204            .header("Authorization", format!("Bearer {}", supabase_key))
205            .send()
206            .await
207        {
208            if let Ok(skills_data) = skills_resp.json::<Vec<serde_json::Value>>().await {
209                result.skills = skills_data
210                    .into_iter()
211                    .filter_map(|s| s.get("skill").and_then(|v| v.as_str()).map(|s| s.to_string()))
212                    .collect();
213            }
214        }
215        
216        // Note: Projects would be fetched similarly if needed
217    }
218    
219    Ok(result)
220}

222/// Scrapes a Devpost profile and reports progress through a callback.
223///
224/// This function navigates to the Devpost profile, extracts skills and interests, and stores the data in the database.
225/// Progress is reported through the provided callback closure.
226///
227/// # Arguments
228/// * `client` - The Fantoccini WebDriver client for browser automation.
229/// * `url` - The URL of the Devpost profile to scrape.
230/// * `progress` - A callback closure for reporting progress messages.
231///
232/// # Returns
233/// * `Result<DevpostScrapeResult>` - The result containing the scrape outcome or an error.
234pub async fn scrape_devpost_profile_with_progress<F>(
235    client: &mut FantocciniClient,
236    url: &str,
237    mut progress: F,
238) -> anyhow::Result<DevpostScrapeResult>
239where
240    F: FnMut(String) + Send,
241{
242    progress("Navigating to Devpost profile...".to_string());
243    if let Err(e) = client.goto(url).await {
244        progress(format!("[error] Failed to navigate: {}", e));
245        return Err(anyhow::anyhow!(e));
246    }
247    sleep(Duration::from_secs(2)).await;
248
249    // Profile image extraction
250    let profile_image_url = match client.find(Locator::Css(".user-profile-photo img")).await {
251        Ok(elem) => match elem.attr("src").await {
252            Ok(Some(src)) => {
253                progress("Found profile image.".to_string());
254                Some(src)
255            },
256            _ => None,
257        },
258        Err(_) => None,
259    };
260
261    // Skills extraction (Devpost: <ul class="portfolio-tags"> <li> <a> or <span.cp-tag>)
262    let mut skills = vec![];
263    if let Ok(li_elements) = client.find_all(Locator::Css("ul.portfolio-tags li")).await {
264        for li in li_elements {
265            // Try to find the <a> inside this <li>
266            if let Ok(a_elem) = li.find(Locator::Css("a")).await {
267                if let Ok(Some(text)) = a_elem.text().await.map(Some) {
268                    let trimmed = text.trim().to_string();
269                    if !trimmed.is_empty() && !skills.contains(&trimmed) {
270                        progress(format!("Collected skill: {}", trimmed));
271                        skills.push(trimmed);
272                        continue;
273                    }
274                }
275            }
276            // If no <a>, try to find the <span> and get its text
277            if let Ok(span_elem) = li.find(Locator::Css("span.cp-tag")).await {
278                if let Ok(Some(text)) = span_elem.text().await.map(Some) {
279                    let trimmed = text.trim().to_string();
280                    if !trimmed.is_empty() && !skills.contains(&trimmed) {
281                        progress(format!("Collected skill: {}", trimmed));
282                        skills.push(trimmed);
283                    }
284                }
285            }
286        }
287    }
288
289    // Interests extraction (Devpost: <div class="tag-list themes"> <ul> <li> <span.cp-tag> or <a>)
290    let mut interests = vec![];
291    if let Ok(div) = client.find(Locator::Css("div.tag-list.themes")).await {
292        if let Ok(li_elements) = div.find_all(Locator::Css("ul li")).await {
293            for li in li_elements {
294                // Try <span class="cp-tag"> (most are like this)
295                if let Ok(span_elem) = li.find(Locator::Css("span.cp-tag")).await {
296                    if let Ok(Some(text)) = span_elem.text().await.map(Some) {
297                        let trimmed = text.trim().to_string();
298                        if !trimmed.is_empty() && !interests.contains(&trimmed) {
299                            progress(format!("Collected interest: {}", trimmed));
300                            interests.push(trimmed);
301                            continue;
302                        }
303                    }
304                }
305                // Try <a> for linked interests (e.g. Machine Learning/AI)
306                if let Ok(a_elem) = li.find(Locator::Css("a")).await {
307                    if let Ok(Some(text)) = a_elem.text().await.map(Some) {
308                        let trimmed = text.trim().to_string();
309                        if !trimmed.is_empty() && !interests.contains(&trimmed) {
310                            progress(format!("Collected interest: {}", trimmed));
311                            interests.push(trimmed);
312                        }
313                    }
314                }
315            }
316        }
317    }
318
319    // --- Insert into Supabase, as in original working logic ---
320    let supabase_url = std::env::var("SUPABASE_URL").expect("SUPABASE_URL not set");
321    let supabase_key = std::env::var("SUPABASE_SERVICE_ROLE_KEY").expect("SUPABASE_SERVICE_ROLE_KEY not set");
322    let http_client = reqwest::Client::new();
323
324    // Get user_uuid using the same lookup logic
325    let lookup = lookup_or_create_user_and_check_scraped(
326        &http_client,
327        &supabase_url,
328        &supabase_key,
329        &std::env::var("CLERK_USER_ID").unwrap_or_default(), // fallback if not present, using old env var name for backward compatibility
330        &std::env::var("CLERK_EMAIL").unwrap_or_default(),
331    ).await;
332    let user_uuid = match lookup {
333        Ok(l) => l.user_uuid,
334        Err(e) => {
335            progress(format!("[error] Could not get user_uuid: {}", e));
336            return Err(anyhow::anyhow!(e));
337        }
338    };
339
340    // Insert profile image if found
341    if let Some(ref image_url) = profile_image_url {
342        let user_query_url = format!("{}/rest/v1/user", supabase_url.trim_end_matches('/'));
343        let update_resp = http_client
344            .patch(&user_query_url)
345            .header("apikey", &supabase_key)
346            .header("Authorization", format!("Bearer {}", supabase_key))
347            .header("Content-Type", "application/json")
348            .query(&[("id".to_string(), format!("eq.{}", user_uuid))])
349            .json(&serde_json::json!({ "devpostprofileimageurl": image_url }))
350            .send()
351            .await;
352        match update_resp {
353            Ok(r) => {
354                let status = r.status();
355                let text = r.text().await.unwrap_or_else(|_| "<no body>".to_string());
356                progress(format!("Supabase user image update: status={} body={}", status, text));
357            }
358            Err(e) => {
359                progress(format!("Supabase user image update error: {:?}", e));
360            }
361        }
362    }
363
364    // Insert all skills as an array in one row
365    let skills_row = serde_json::json!({
366        "user_uuid": user_uuid,
367        "skills": skills,
368    });
369    let skills_resp = http_client.post(format!("{}/rest/v1/devpost_skills", supabase_url.trim_end_matches('/')))
370        .header("apikey", &supabase_key)
371        .header("Authorization", format!("Bearer {}", supabase_key))
372        .header("Content-Type", "application/json")
373        .json(&skills_row)
374        .send()
375        .await;
376    match skills_resp {
377        Ok(r) => {
378            let status = r.status();
379            let text = r.text().await.unwrap_or_else(|_| "<no body>".to_string());
380            if status == 409 && text.contains("unique_user_uuid_skills") {
381                progress("User Skills already stored.".to_string());
382            } else if status == 201 {
383                progress("User Skills imported successfully.".to_string());
384            } else {
385                progress(format!("Supabase skills insert: status={} body={}", status, text));
386            }
387        }
388        Err(e) => {
389            progress(format!("Supabase skills insert error: {:?}", e));
390        }
391    }
392
393    // Insert all interests as an array in one row
394    let interests_row = serde_json::json!({
395        "user_uuid": user_uuid,
396        "interests": interests,
397    });
398    let interests_resp = http_client.post(format!("{}/rest/v1/devpost_interests", supabase_url.trim_end_matches('/')))
399        .header("apikey", &supabase_key)
400        .header("Authorization", format!("Bearer {}", supabase_key))
401        .header("Content-Type", "application/json")
402        .json(&interests_row)
403        .send()
404        .await;
405    match interests_resp {
406        Ok(r) => {
407            let status = r.status();
408            let text = r.text().await.unwrap_or_else(|_| "<no body>".to_string());
409            if status == 409 && text.contains("unique_user_uuid_interests") {
410                progress("User Interests already stored.".to_string());
411            } else if status == 201 {
412                progress("User Interests imported successfully.".to_string());
413            } else {
414                progress(format!("Supabase interests insert: status={} body={}", status, text));
415            }
416        }
417        Err(e) => {
418            progress(format!("Supabase interests insert error: {:?}", e));
419        }
420    }
421
422    progress("Finished scraping Devpost profile.".to_string());
423    Ok(DevpostScrapeResult {
424        profile_image_url,
425        skills,
426        interests,
427        message: None,
428    })
429}
