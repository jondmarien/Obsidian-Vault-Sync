1//! # Webhook Service
2//!
3//! This module provides functionality for handling webhook events and integrating with
4//! external services like Clerk (authentication) and Discord. It handles user lifecycle
5//! events and can send notifications to Discord webhooks.
6//!
7//! ## Features
8//! - Process Clerk webhook events (user created/updated/deleted)
9//! - Synchronize user data with Supabase
10//! - Send notifications to Discord webhooks
11//! - Comprehensive error handling and logging
12//!
13//! ## Prerequisites
14//! - `SUPABASE_URL`: URL of the Supabase instance
15//! - `SUPABASE_SERVICE_ROLE_KEY` or `SUPABASE_API_KEY`: API key for Supabase
16//! - Webhook URLs for external services (Discord, etc.)
17

41///
42/// # #[tokio::main]
43/// # async fn main() -> Result<(), Box<dyn std::error::Error>> {

123    /// Handles a Clerk 'user created' webhook event by creating a new user in Supabase.
124    ///
125    /// This method processes a user creation event from Clerk, extracts the relevant
126    /// user information, and creates a corresponding user record in Supabase.
127    ///
128    /// # Arguments
129    /// * `data` - The `UserCreatedData` containing the user information from Clerk
130    ///
131    /// # Returns
132    /// * `Result<WebhookResponse, String>` - A result containing either a success response
133    ///   with a success message or an error message if the operation fails
134    ///
135    /// # Errors
136    /// Returns an error if:
137    /// - The user data is missing an email address
138    /// - The request to Supabase fails
139    /// - The response from Supabase indicates an error
140    ///
141    /// # Example
142    /// ```no_run
143    /// # use judgeflow_backend::services::webhook_service::WebhookService;
144    /// # use judgeflow_backend::route_models::webhook_model::{UserCreatedData, EmailAddress};
145    /// # #[tokio::main]
146    /// # async fn main() -> Result<(), Box<dyn std::error::Error>> {
147    /// let service = WebhookService::new();
148    /// let user_data = UserCreatedData {
149    ///     id: "user_123".to_string(),
150    ///     email_addresses: vec![EmailAddress {
151    ///         email_address: "user@example.com".to_string(),
152    ///         // ... other fields ...
153    ///     }],
154    ///     first_name: Some("John".to_string()),
155    ///     last_name: Some("Doe".to_string()),
156    ///     username: Some("johndoe".to_string()),
157    ///     // ... other fields ...
158    /// };
159    ///
160    /// match service.create_user(user_data).await {
161    ///     Ok(response) => println!("User created: {}", response.message),
162    ///     Err(e) => eprintln!("Failed to create user: {}", e),
163    /// }
164    /// # Ok(())
165    /// # }
166    /// ```
167    pub async fn create_user(&self, data: UserCreatedData) -> Result<WebhookResponse, String> {
168        let client = Client::new();
169        
170        // Extract user details from the webhook event
171        let email = data.email_addresses
172            .get(0)
173            .map(|e| e.email_address.clone())
174            .ok_or("No email address found".to_string())?;
175
176        let first_name = data.first_name.clone().unwrap_or_else(|| "".to_string());
177        let last_name = data.last_name.clone();
178
179        let username = data.username.clone().unwrap_or_else(|| "".to_string());
180
181        use uuid::Uuid;
182        // Generate a new UUID for the user
183        let user_id = Uuid::new_v4();
184
185        // Build the payload for Supabase
186        let payload = json!({
187            "id": user_id,
188            "email": email,
189            "first_name": first_name,
190            "last_name": last_name,
191            "username": username,
192            "clerk_uid": data.id,
193            "hackathon_ids": serde_json::Value::Null,
194            // created_at omitted for default
195        });
196
197        // Insert user into Supabase
198        let resp = client
199            .post(format!("{}/rest/v1/user", self.supabase_url.trim_end_matches('/')))
200            .header("apikey", &self.supabase_key)
201            .header("Authorization", format!("Bearer {}", self.supabase_key))
202            .header("Content-Type", "application/json")
203            .header("Prefer", "return=representation")
204            .json(&payload)
205            .send()
206            .await
207            .map_err(|e| format!("Failed to contact Supabase: {}", e))?;
208
209        if resp.status().is_success() {
210            Ok(WebhookResponse {
211                success: true,
212                message: "User successfully created".to_string(),
213            })
214        } else {
215            let error_text = resp.text().await.unwrap_or_default();
216            tracing::error!("Supabase insert failed: {}", error_text);
217            Err(format!("Supabase insert failed: {}", error_text))
218        }
219    }

416    /// Send a message to a Discord webhook
417    ///
418    /// # Arguments
419    /// * `webhook_url` - The Discord webhook URL
420    /// * `message` - The message to send to Discord
421    ///
422    /// # Returns
423    /// * `Result<WebhookResponse, String>` - The result of the Discord webhook operation
424    pub async fn send_discord_webhook(&self, webhook_url: &str, message: DiscordMessage) -> Result<WebhookResponse, String> {
425        // Validate the Discord message
426        message.validate().map_err(|e| e.to_string())?;
427
428        let client = Client::new();
429        let payload = json!({ "content": message.content });
430        
431        let resp = client
432            .post(webhook_url)
433            .json(&payload)
434            .send()
435            .await
436            .map_err(|e| format!("Failed to send Discord webhook: {}", e))?;
437
438        if resp.status().is_success() {
439            Ok(WebhookResponse {
440                success: true,
441                message: "Discord message sent successfully".to_string(),
442            })
443        } else {
444            let error_text = resp.text().await.unwrap_or_default();
445            tracing::error!("Discord webhook failed: {}", error_text);
446            Err(format!("Discord webhook failed: {}", error_text))
447        }
448    }
