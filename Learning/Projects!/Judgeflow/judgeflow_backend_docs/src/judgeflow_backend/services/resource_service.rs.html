1//! # Resource Service
2//!
3//! This module provides a high-level interface for interacting with Supabase database resources.
4//! It handles CRUD (Create, Read, Update, Delete) operations for various resources like
5//! hackathons, screenings, and submissions, with built-in validation.
6//!
7//! ## Features
8//! - CRUD operations for database resources
9//! - Built-in request validation
10//! - Environment-based configuration
11//! - Error handling and status code mapping
12//!
13//! ## Configuration
14//! The service requires the following environment variables:
15//! - `SUPABASE_URL`: The URL of your Supabase instance
16//! - `SUPABASE_SERVICE_KEY`: The service role key for Supabase
17//!
18//! ## Example
19//! ```no_run
20//! use uuid::Uuid;
21//! use serde_json::json;
22//! use judgeflow_backend::services::resource_service::ResourceService;
23//!
24//! # #[tokio::main]
25//! # async fn main() -> Result<(), Box<dyn std::error::Error>> {
26//! let service = ResourceService::new();
27//! let user_id = Uuid::new_v4();
28//!
29//! // Create a new item
30//! let new_item = json!({ "name": "Example", "user_id": user_id });
31//! let created = service.create_item("resources", new_item).await?;
32//! println!("Created item: {:?}", created);
33//! # Ok(())
34//! # }
35//! ```

43/// A client for interacting with Supabase database resources.
44///
45/// This struct provides methods to perform CRUD operations on various resources
46/// in a Supabase database. It handles authentication, request building, and
47/// response parsing.
48///
49/// # Fields
50/// - `supabase_url`: The base URL of the Supabase instance
51/// - `service_key`: The service role key for authenticating with Supabase
52///
53/// # Panics
54/// The `new` function will panic if the required environment variables are not set.
55pub struct ResourceService {
56    supabase_url: String,
57    service_key: String,
58}
59
60impl ResourceService {
61    /// Creates a new instance of `ResourceService` using environment variables.
62    ///
63    /// # Environment Variables
64    /// - `SUPABASE_URL`: The URL of the Supabase instance
65    /// - `SUPABASE_SERVICE_KEY`: The service role key for Supabase
66    ///
67    /// # Panics
68    /// Panics if either of the required environment variables are not set.
69    ///
70    /// # Example
71    /// ```no_run
72    /// use judgeflow_backend::services::resource_service::ResourceService;
73    ///
74    /// // These environment variables must be set in the environment
75    /// // or in a .env file loaded by dotenv
76    /// let service = ResourceService::new();
77    /// ```
78    pub fn new() -> Self {
79        let supabase_url = env::var("SUPABASE_URL")
80            .expect("SUPABASE_URL environment variable not set");
81            
82        let service_key = env::var("SUPABASE_SERVICE_KEY")
83            .expect("SUPABASE_SERVICE_KEY environment variable not set");
84        
85        Self {
86            supabase_url,
87            service_key,
88        }
89    }

91    /// Retrieves all items from a specified resource table filtered by user ID.
92    ///
93    /// # Arguments
94    /// * `resource` - The name of the resource table to query (e.g., "hackathons", "submissions")
95    /// * `user_id` - The UUID of the user to filter results by
96    ///
97    /// # Returns
98    /// * `Ok(Value)` - A JSON value containing the retrieved items on success
99    /// * `Err(String)` - An error message if the request fails
100    ///
101    /// # Errors
102    /// Returns an error if:
103    /// - The HTTP request fails
104    /// - The response cannot be parsed as JSON
105    /// - The Supabase API returns a non-success status code
106    ///
107    /// # Example
108    /// ```no_run
109    /// # use uuid::Uuid;
110    /// # use judgeflow_backend::services::resource_service::ResourceService;
111    /// # #[tokio::main]
112    /// # async fn main() -> Result<(), Box<dyn std::error::Error>> {
113    /// let service = ResourceService::new();
114    /// let user_id = Uuid::parse_str("123e4567-e89b-12d3-a456-426614174000")?;
115    /// 
116    /// match service.get_items("hackathons", &user_id).await {
117    ///     Ok(items) => println!("Found hackathons: {:?}", items),
118    ///     Err(e) => eprintln!("Error retrieving hackathons: {}", e),
119    /// }
120    /// # Ok(())
121    /// # }
122    /// ```
123    pub async fn get_items(&self, resource: &str, user_id: &Uuid) -> Result<Value, String> {
124        let client = reqwest::Client::new();
125        let url = format!(
126            "{}/{}?user_id=eq.{}",
127            self.supabase_url.trim_end_matches('/'),
128            resource,
129            user_id
130        );
131
132        let response = client
133            .get(&url)
134            .header("apikey", &self.service_key)
135            .header("Authorization", format!("Bearer {}", &self.service_key))
136            .send()
137            .await
138            .map_err(|e| format!("Request failed: {}", e))?;
139            
140        let status = response.status();
141        if status.is_success() {
142            response.json().await.map_err(|e| format!("Failed to parse response: {}", e))
143        } else {
144            Err(format!("API request failed with status: {}", status))
145        }
146    }

148    /// Creates a new item in the specified resource table.
149    ///
150    /// # Arguments
151    /// * `resource` - The name of the resource table (e.g., "hackathons", "submissions")
152    /// * `body` - A JSON value containing the data for the new item
153    ///
154    /// # Returns
155    /// * `Ok(Value)` - The created item as a JSON value on success
156    /// * `Err(String)` - An error message if the creation fails
157    ///
158    /// # Errors
159    /// Returns an error if:
160    /// - The HTTP request fails
161    /// - The response cannot be parsed as JSON
162    /// - The Supabase API returns a non-success status code
163    ///
164    /// # Example
165    /// ```no_run
166    /// # use serde_json::json;
167    /// # use uuid::Uuid;
168    /// # use judgeflow_backend::services::resource_service::ResourceService;
169    /// # #[tokio::main]
170    /// # async fn main() -> Result<(), Box<dyn std::error::Error>> {
171    /// let service = ResourceService::new();
172    /// let user_id = Uuid::new_v4();
173    /// 
174    /// let new_hackathon = json!({
175    ///     "name": "My Hackathon",
176    ///     "description": "An amazing hackathon!",
177    ///     "user_id": user_id.to_string(),
178    ///     "start_date": "2024-01-01T00:00:00Z",
179    ///     "end_date": "2024-01-02T00:00:00Z"
180    /// });
181    ///
182    /// match service.create_item("hackathons", new_hackathon).await {
183    ///     Ok(created) => println!("Created: {:?}", created),
184    ///     Err(e) => eprintln!("Error creating item: {}", e),
185    /// }
186    /// # Ok(())
187    /// # }
188    /// ```
189    pub async fn create_item(&self, resource: &str, body: Value) -> Result<Value, String> {
190        let client = reqwest::Client::new();
191        let url = format!(
192            "{}/{}",
193            self.supabase_url.trim_end_matches('/'),
194            resource
195        );
196
197        // Validate the resource before sending the request
198        if let Err(e) = self.validate_resource(resource, &body) {
199            return Err(format!("Validation failed: {}", e));
200        }
201
202        let response = client
203            .post(&url)
204            .header("apikey", &self.service_key)
205            .header("Authorization", format!("Bearer {}", &self.service_key))
206            .header("Content-Type", "application/json")
207            .header("Prefer", "return=representation") // Return the created record
208            .json(&body)
209            .send()
210            .await
211            .map_err(|e| format!("Request failed: {}", e))?;
212            
213        let status = response.status();
214        if status.is_success() {
215            response.json().await.map_err(|e| format!("Failed to parse response: {}", e))
216        } else {
217            let error_msg = match response.text().await {
218                Ok(body) => format!("API request failed with status {}: {}", status, body),
219                Err(_) => format!("API request failed with status: {}", status),
220            };
221            Err(error_msg)
222        }
223    }

225    /// Updates an existing item in the specified resource table.
226    ///
227    /// # Arguments
228    /// * `resource` - The name of the resource table (e.g., "hackathons", "submissions")
229    /// * `id` - The ID of the item to update
230    /// * `user_id` - The UUID of the user who owns the item (for authorization)
231    /// * `body` - A JSON value containing the fields to update
232    ///
233    /// # Returns
234    /// * `Ok(Value)` - The updated item as a JSON value on success
235    /// * `Err(String)` - An error message if the update fails
236    ///
237    /// # Errors
238    /// Returns an error if:
239    /// - The HTTP request fails
240    /// - The response cannot be parsed as JSON
241    /// - The Supabase API returns a non-success status code
242    /// - The user is not authorized to update the item
243    ///
244    /// # Example
245    /// ```no_run
246    /// # use serde_json::json;
247    /// # use uuid::Uuid;
248    /// # use judgeflow_backend::services::resource_service::ResourceService;
249    /// # #[tokio::main]
250    /// # async fn main() -> Result<(), Box<dyn std::error::Error>> {
251    /// let service = ResourceService::new();
252    /// let user_id = Uuid::parse_str("123e4567-e89b-12d3-a456-426614174000")?;
253    /// let item_id = "abc123";
254    /// 
255    /// let updates = json!({
256    ///     "description": "Updated description for the hackathon"
257    /// });
258    ///
259    /// match service.update_item("hackathons", item_id, &user_id, updates).await {
260    ///     Ok(updated) => println!("Updated: {:?}", updated),
261    ///     Err(e) => eprintln!("Error updating item: {}", e),
262    /// }
263    /// # Ok(())
264    /// # }
265    /// ```
266    pub async fn update_item(&self, resource: &str, id: &str, user_id: &Uuid, body: Value) -> Result<Value, String> {
267        let client = reqwest::Client::new();
268        let url = format!(
269            "{}/{}?id=eq.{}&user_id=eq.{}",
270            self.supabase_url.trim_end_matches('/'),
271            resource,
272            id,
273            user_id
274        );
275
276        // Validate the resource before sending the request
277        if let Err(e) = self.validate_resource(resource, &body) {
278            return Err(format!("Validation failed: {}", e));
279        }
280
281        let response = client
282            .patch(&url)
283            .header("apikey", &self.service_key)
284            .header("Authorization", format!("Bearer {}", &self.service_key))
285            .header("Content-Type", "application/json")
286            .header("Prefer", "return=representation") // Return the updated record
287            .json(&body)
288            .send()
289            .await
290            .map_err(|e| format!("Request failed: {}", e))?;
291            
292        let status = response.status();
293        if status.is_success() {
294            response.json().await.map_err(|e| format!("Failed to parse response: {}", e))
295        } else if status == StatusCode::NOT_FOUND {
296            Err("Item not found or not owned by user".to_string())
297        } else {
298            let error_msg = match response.text().await {
299                Ok(body) => format!("API request failed with status {}: {}", status, body),
300                Err(_) => format!("API request failed with status: {}", status),
301            };
302            Err(error_msg)
303        }
304    }

374    /// Validates resource data against the appropriate model.
375    ///
376    /// This is an internal method used to validate data before it's sent to the database.
377    /// It ensures that the data conforms to the expected schema for the resource.
378    ///
379    /// # Arguments
380    /// * `resource` - The name of the resource being validated
381    /// * `body` - The JSON data to validate
382    ///
383    /// # Returns
384    /// * `Ok(())` - If the data is valid
385    /// * `Err(String)` - A validation error message if the data is invalid
386    ///
387    /// # Note
388    /// Currently supports validation for:
389    /// - "hackathon" - Validates against the `Hackathon` model
390    /// - "screenings" - Validates against the `Screening` model
391    /// - "submissions" - Validates against the `Submission` model
392    /// Other resource types will pass validation without checking.
393    ///
394    /// TODO: Replace with a typed enum for the resource parameter
395    pub fn validate_resource(&self, resource: &str, body: &Value) -> Result<(), String> {
396        match resource {
397            "hackathon" => {
398                serde_json::from_value::<Hackathon>(body.clone())
399                    .map_err(|e| e.to_string())?
400                    .validate()
401                    .map_err(|e| e.to_string())
402            },
403            "screenings" => {
404                serde_json::from_value::<Screening>(body.clone())
405                    .map_err(|e| e.to_string())?
406                    .validate()
407                    .map_err(|e| e.to_string())
408            },
409            "submissions" => {
410                serde_json::from_value::<Submission>(body.clone())
411                    .map_err(|e| e.to_string())?
412                    .validate()
413                    .map_err(|e| e.to_string())
414            },
415            _ => Ok(()), // No validation for other resources
416        }
417    }
