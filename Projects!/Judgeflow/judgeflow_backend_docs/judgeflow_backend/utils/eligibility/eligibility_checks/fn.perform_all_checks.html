Expand description

Performs all standard eligibility checks based on the provided configuration.

This function runs multiple eligibility checks in parallel and returns a vector of all results. The checks performed are determined by the configuration provided.

ยงArguments

  • input - The eligibility check input containing participant data
  • config - Configuration specifying which checks to perform and their parameters

ยงReturns

A Result containing a vector of EligibilityCriterionResult with the outcome of each check.

ยงExample

// Create a participant
let input = EligibilityCheckInput {
    participant_name: "Jordan Lee".to_string(),
    participant_email: "jordan@example.com".to_string(),
    participant_age: Some(22),
    participant_country: Some("Canada".to_string()),
    participant_employment_status: Some("Student".to_string()),
    additional_data: HashMap::from([
        ("available_equipment".to_string(), "Laptop, Headphones".to_string()),
        ("previous_wins".to_string(), "1".to_string()),
        ("available_hours".to_string(), "20".to_string())
    ]),
};

// Configure which checks to perform
let config = EligibilityConfig {
    check_minimum_age: Some(18),
    check_eligible_countries: Some(vec!["Canada".to_string(), "USA".to_string()]),
    check_employment_status: Some(vec!["Student".to_string(), "Unemployed".to_string()]),
    check_previous_wins: Some(3), // Max 3 wins allowed
    check_time_commitment: Some(15), // At least 15 hours required
    check_equipment_requirements: Some(vec!["Laptop".to_string()]),
    ..Default::default()
};

// Perform all configured checks
let results = perform_all_checks(&input, &config).await?;

// Check that we got the expected number of results
assert_eq!(results.len(), 5);

// Verify that all checks passed
assert!(results.iter().all(|r| r.status.is_eligible()));
